# The type of the plane determines its most basic attributes and traits.
# Material: the standard plane type. It can have many different forms.
# Elemental: A plane where a single element dominates everything.
# Ethereal: An ethereal plane is overlayed over a material world, but is dominated by an element. It copies most traits and attributes of the material plane, but can have a number of element specific differences.
plane_types = { 
	"material" "elemental" "ethereal"
}

## The size of a plane determines with how many other planes it can connect and how much space there is inside of it.
# inifinte: Can connect to any number of planes.
# large: can connect with up to 16 planes
# medium: can connect with up to 8 planes
# small: can connect with up to 4 planes
# pocket: can only connect to a single plane.

	"pocket" "small" "medium" "large" "infinite"
